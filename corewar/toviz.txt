typedef struct	s_vm
{
	t_plr_ardata	*plrdata;		// см ниже
	size_t			cyclen;			// общее количество циклов
	int				liven;			// общее количество live до проверки
	int				cycles_to_die;	// количество циклов до проверки
	int				checkn;			// количество проверок (по идее можно не включать в вывод)
	unsigned char	*field;			// само поле
	size_t			field_size;		// размер поля
	t_prcs			*prcs;			// скорее всего ты и так используешь эту структуру по полной
									// но для вывода prcs->livecycle - цикл, в котором текущий процесс последний раз сообщил что он жив
}				t_vm;

typedef struct	s_plr_ardata
{
	int					uid; 		// Номер игрока (хранится не в отрицательном значении)
	char				*name;		// Имя
	char				*data;		// Начальный код игрока (скорее всео не понадобится)
	size_t				liven;		// Количество live выбранного игрока до проверки
	size_t				data_size;	// Размер начального кода игрока (скорее всего тоже не понадобится)
	struct s_plr_ardata	*next;
}				t_plr_ardata;
